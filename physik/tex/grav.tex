\section{Gravitation}

\begin{boxleft}\bla{Gravitationsgesetzt}
\des[\newton]{F_g}{Gravitationskraft}\\
\des[\newton\meter\tothe{2}\per\kilo\gram\tothe{2}]{G}{Gravitationskonstante}\\
\des[\meter]{r_{12}}{Schwerpunktabstand der Körper}\\
\des[\kilo\gram]{m_i}{Masse des Körper i}\\
\des[\meter\per\second\tothe{2}]{E_g}{Gravitationsfeldstärke}
\end{boxleft}\begin{boxrightshaded}
\begin{align}
\vv{F}_{g,2}&=-G\frac{m_1m_2}{r_{12}^2}\vv{e}_r\\
\vv{F}_g&=\vv{E}_g\cdot m=\vv{g}m
\end{align}
\end{boxrightshaded}

\begin{boxleft}\bla{Gravitationspotenzial}
\des[\joule\per\kilo\gram]{\phi}{Gravitationspotenzial}\\
\des[\newton\meter\tothe{2}\per\kilo\gram\tothe{2}]{G}{Gravitationskonstante}\\
\des[\meter]{r}{Abstand der anziehenden Kraft}\\
\des[\kilo\gram]{M}{Masse des Anziehende Körpers}
\end{boxleft}\begin{boxrightshaded}
\begin{align}
\phi&=-G\frac{M}{r}\\
\vv{E}_g&=\grad\phi
\end{align}
\end{boxrightshaded}

\begin{boxleft}\bla{Arbeit}
\des[\joule\per\kilo\gram]{\phi}{Gravitationspotenzial}\\

\end{align}
\end{boxrightshaded}
